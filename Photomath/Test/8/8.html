(f\circ\text{g})(x)=\sqrt{\textcolor{mathSequenceLatex}{g(x)}}

(f\circ\text{g})(x)=\sqrt{\textcolor{mathSequenceLatex}{x+2}}

(g\circ\text{f})(x)=g(\textcolor{mathSequenceLatex}{f(x)})
